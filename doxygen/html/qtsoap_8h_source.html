77 class QtSmartPtr
78 {
79 public:
80  inline QtSmartPtr(T* data = 0) {
81  d = data;
82  r = new int;
83  *r = 1;
84  }
85 
86  inline QtSmartPtr(const QtSmartPtr& copy) {
87  if (*copy.r != 0)
88  ++(*copy.r);
89 
90  r = copy.r;
91  d = copy.d;
92  }
93 
94  inline ~QtSmartPtr() {
95  if ((*r) == 0)
96  delete r;
97  else if ((*r) != 0 && --(*r) == 0) {
98  delete r;
99 
100  if (d)
101  delete d;
102  }
103  }
104 
105  inline QtSmartPtr& operator =(const QtSmartPtr& copy) {
106  if (*copy.r != 0)
107  ++(*copy.r);
108 
109  if ((*r) == 0)
110  delete r;
111  else if ((*r) != 0 && --(*r) == 0) {
112  delete r;
113 
114  if (d)
115  delete d;
116  }
117 
118  r = copy.r;
119  d = copy.d;
120  return *this;
121  }
122 
123  inline T& operator *() const {
124  return *d;
125  }
126 
127  inline T* operator ->() const {
128  return d;
129  }
130 
131  inline T* ptr() const {
132  return d;
133  }
134 
135  inline T& ref() const {
136  return *d;
137  }
138 
139  inline T* releasedPtr() const {
140  (*r) = 0;
141  return d;
142  }
143 
144  inline bool isNull() const {
145  return d == 0;
146  }
147 
148 private:
149  int* r;
150  T* d;
151 };
